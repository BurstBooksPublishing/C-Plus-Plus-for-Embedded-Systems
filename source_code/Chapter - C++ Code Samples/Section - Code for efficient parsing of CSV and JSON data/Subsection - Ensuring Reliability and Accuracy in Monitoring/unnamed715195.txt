using json = nlohmann::json;

// Function to parse CSV data
std::vector> parseCSV(const std::string& filename) {
    std::vector> data;
    std::ifstream file(filename);
    std::string line;

    while (std::getline(file, line)) {
        std::vector row;
        std::stringstream ss(line);
        std::string cell;

        while (std::getline(ss, cell, ',')) {
            row.push_back(cell);
        }
        data.push_back(row);
    }
    return data;
}

// Function to parse JSON data
json parseJSON(const std::string& filename) {
    std::ifstream file(filename);
    json data;
    file >> data;
    return data;
}

int main() {
    // Example usage for CSV parsing
    std::string csvFile = "data.csv";
    auto csvData = parseCSV(csvFile);
    std::cout << "CSV Data Parsed Successfully!" << std::endl;

    // Example usage for JSON parsing
    std::string jsonFile = "data.json";
    auto jsonData = parseJSON(jsonFile);
    std::cout << "JSON Data Parsed Successfully!" << std::endl;

    return 0;
}
