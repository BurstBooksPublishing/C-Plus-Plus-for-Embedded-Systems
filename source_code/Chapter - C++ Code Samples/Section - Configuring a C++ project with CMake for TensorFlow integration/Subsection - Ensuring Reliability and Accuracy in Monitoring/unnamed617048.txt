# CMakeLists.txt for TensorFlow C++ integration

cmake_minimum_required(VERSION 3.14)  # Minimum CMake version required
project(TensorFlowCppIntegration)     # Project name

# Set C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find TensorFlow package
find_package(TensorFlow REQUIRED)

# Include TensorFlow headers
include_directories(${TensorFlow_INCLUDE_DIRS})

# Add executable
add_executable(TensorFlowExample main.cpp)

# Link TensorFlow libraries
target_link_libraries(TensorFlowExample ${TensorFlow_LIBRARIES})
