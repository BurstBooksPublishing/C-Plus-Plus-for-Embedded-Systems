uint8_t computeChecksum(const uint8_t* data, size_t length) {
    uint8_t checksum = 0;
    for (size_t i = 0; i < length; ++i) {
        checksum ^= data[i];
    }
    return checksum;
}

bool verifyData(const uint8_t* data, size_t length, uint8_t expectedChecksum) {
    return computeChecksum(data, length) == expectedChecksum;
}
