void bootloader() {
    // Initialize system peripherals
    init_system();

    // Copy application from Flash to RAM
    uint32_t* src = reinterpret_cast(0x08000000); // Flash start
    uint32_t* dest = reinterpret_cast(0x20000000); // RAM start
    for (int i = 0; i < APP_SIZE; ++i) {
        dest[i] = src[i];
    }

    // Jump to application
    void (*app_entry)() = reinterpret_cast(0x20000000);
    app_entry();
}
