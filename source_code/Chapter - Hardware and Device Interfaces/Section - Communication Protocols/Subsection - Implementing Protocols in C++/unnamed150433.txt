#include 

void spi_init() {
    // Set MOSI and SCK as output, others as input
    DDRB |= (1 << DDB3) | (1 << DDB5);
    // Enable SPI, Master, set clock rate fck/16
    SPCR = (1 << SPE) | (1 << MSTR) | (1 << SPR0);
}

void spi_send(uint8_t data) {
    // Start transmission
    SPDR = data;
    // Wait for transmission complete
    while (!(SPSR & (1 << SPIF)));
}
