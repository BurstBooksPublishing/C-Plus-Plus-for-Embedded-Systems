#include 

int main() {
    DDRB &= ~(1 << DDB1); // Set PB1 as input
    PORTB |= (1 << PB1); // Enable pull-up resistor on PB1
    while (1) {
        if (!(PINB & (1 << PINB1))) { // Check if button is pressed
            PORTB |= (1 << PB0); // Turn on LED on PB0
        } else {
            PORTB &= ~(1 << PB0); // Turn off LED on PB0
        }
    }
}
