#include 

void configureGPIO() {
    // Enable GPIOB clock
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOBEN;

    // Configure PB0 as output
    GPIOB->MODER &= ~(0b11 << (0 * 2)); // Clear mode bits for PB0
    GPIOB->MODER |= (0b01 << (0 * 2));  // Set mode to output for PB0
}

void toggleLED() {
    // Toggle PB0
    GPIOB->ODR ^= (1 << 0);
}

int main() {
    configureGPIO();
    while (1) {
        toggleLED();
        for (volatile int i = 0; i < 1000000; i++); // Delay
    }
}
