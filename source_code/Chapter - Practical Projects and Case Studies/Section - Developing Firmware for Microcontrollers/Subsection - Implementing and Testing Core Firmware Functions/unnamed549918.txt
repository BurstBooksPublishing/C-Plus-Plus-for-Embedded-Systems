void UART_Init() {
    // Enable UART2 clock
    RCC->APB1ENR |= RCC_APB1ENR_USART2EN;

    // Set baud rate to 9600 (assuming 84 MHz clock)
    USART2->BRR = 546;

    // Enable transmitter and receiver
    USART2->CR1 |= USART_CR1_TE | USART_CR1_RE;

    // Enable UART2
    USART2->CR1 |= USART_CR1_UE;
}
