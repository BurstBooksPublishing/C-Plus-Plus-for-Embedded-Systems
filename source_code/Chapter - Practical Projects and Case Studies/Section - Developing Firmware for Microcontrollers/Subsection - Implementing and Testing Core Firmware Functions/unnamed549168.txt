void Timer_Init() {
    // Enable Timer 2 clock
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;

    // Set prescaler to 84,000 for 1 ms tick (assuming 84 MHz clock)
    TIM2->PSC = 83999;

    // Set auto-reload register for 1000 ms (1 second)
    TIM2->ARR = 999;

    // Enable Timer 2
    TIM2->CR1 |= TIM_CR1_CEN;
}
