try {
    // Attempt to read sensor data
    float sensorValue = readSensor();
    if (!validateSensorData(sensorValue, 0.0f, 100.0f)) {
        throw std::runtime_error("Invalid sensor reading");
    }
} catch (const std::exception& e) {
    // Log the error and take corrective action
    logError(e.what());
    takeCorrectiveAction();
}
